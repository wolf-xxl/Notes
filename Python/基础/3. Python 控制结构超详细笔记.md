## 1. 条件语句基础

### 1.1 基本的if语句
```python
# 最简单的if语句
age = 18
if age >= 18:
    print("你已经成年了")  # 输出：你已经成年了

# if-else语句
age = 16
if age >= 18:
    print("你已经成年了")
else:
    print("你还未成年")  # 输出：你还未成年

# if-elif-else语句
score = 85
if score >= 90:
    print("优秀")
elif score >= 80:
    print("良好")  # 输出：良好
elif score >= 70:
    print("中等")
else:
    print("不及格")
```

### 1.2 条件表达式详解
```python
# 条件表达式中的比较运算符
a, b = 10, 20

print("a == b:", a == b)  # 输出：a == b: False
print("a != b:", a != b)  # 输出：a != b: True
print("a < b:", a < b)    # 输出：a < b: True
print("a > b:", a > b)    # 输出：a > b: False
print("a <= b:", a <= b)  # 输出：a <= b: True
print("a >= b:", a >= b)  # 输出：a >= b: False

# 身份运算符
x = [1, 2, 3]
y = [1, 2, 3]
z = x

print("x is y:", x is y)    # 输出：x is y: False
print("x is z:", x is z)    # 输出：x is z: True
print("x is not y:", x is not y)  # 输出：x is not y: True

# 成员运算符
fruits = ["apple", "banana", "cherry"]
print("'apple' in fruits:", "apple" in fruits)        # 输出：'apple' in fruits: True
print("'orange' not in fruits:", "orange" not in fruits)  # 输出：'orange' not in fruits: True
```

```python
    elif user_input == "yes":
        print("用户选择了是")
    elif user_input == "no":
```

### 4.3 列表推导式

### 6.2 try-except-else-finally结构
```python
# 完整的异常处理结构
def divide_numbers(a, b):
    try:
        result = a / b
    except ZeroDivisionError:
        print("错误：除数不能为零")
        return None
    except TypeError:
        print("错误：类型不正确")
        return None
    else:
        print("计算成功完成")
        return result
    finally:
        print("这是finally块，总是会执行")

print("正常情况:")
result1 = divide_numbers(10, 2)  # 输出：计算成功完成\n这是finally块，总是会执行
print("结果:", result1)  # 输出：结果: 5.0

print("\n除零错误:")
result2 = divide_numbers(10, 0)  # 输出：错误：除数不能为零\n这是finally块，总是会执行
print("结果:", result2)  # 输出：结果: None

print("\n类型错误:")
result3 = divide_numbers(10, "2")  # 输出：错误：类型不正确\n这是finally块，总是会执行
print("结果:", result3)  # 输出：结果: None
```

## 7. 易错点和难点解析

### 7.1 循环中修改列表的问题
```python
# 错误示例：在循环中修改列表
numbers = [1, 2, 3, 4, 5]
print("原始列表:", numbers)  # 输出：原始列表: [1, 2, 3, 4, 5]

# 错误的做法：在循环中删除元素
# for num in numbers:
#     if num % 2 == 0:
#         numbers.remove(num)
# print("错误的结果:", numbers)  # 可能输出：[1, 3, 5]，但行为不可预测

# 正确的做法1：创建副本
numbers = [1, 2, 3, 4, 5]
for num in numbers[:]:  # 使用切片创建副本
    if num % 2 == 0:
        numbers.remove(num)
print("正确结果1:", numbers)  # 输出：正确结果1: [1, 3, 5]

# 正确的做法2：使用列表推导式
numbers = [1, 2, 3, 4, 5]
numbers = [num for num in numbers if num % 2 != 0]
print("正确结果2:", numbers)  # 输出：正确结果2: [1, 3, 5]

# 正确的做法3：反向遍历
numbers = [1, 2, 3, 4, 5]
for i in range(len(numbers)-1, -1, -1):
    if numbers[i] % 2 == 0:
        del numbers[i]
print("正确结果3:", numbers)  # 输出：正确结果3: [1, 3, 5]
```

## 8. 高级技巧和最佳实践

### 8.1 使用any()和all()函数
```python
# any(): 只要有一个为True就返回True
numbers = [0, 1, 0, 0]
print("any([0, 1, 0, 0]):", any(numbers))  # 输出：any([0, 1, 0, 0]): True

# all(): 所有元素都为True才返回True
numbers = [1, 1, 0, 1]
print("all([1, 1, 0, 1]):", all(numbers))  # 输出：all([1, 1, 0, 1]): False

# 实际应用
scores = [85, 92, 78, 90]
```

```python
class Timer:
    def __init__(self, name):
        self.name = name
    
    def __enter__(self):
        import time
        self.start = time.time()
        return self
    
    def __exit__(self, exc_type, exc_val, exc_tb):
        import time
        self.end = time.time()
        print(f"{self.name} 执行时间: {self.end - self.start:.4f}秒")

# 使用上下文管理器
with Timer("测试代码块"):
    total = 0
    for i in range(1000000):
        total += i
# 输出：测试代码块 执行时间: 0.0456秒

# 文件操作的上下文管理器（自动关闭文件）
with open('example.txt', 'w') as f:
    f.write("Hello, World!")

# 等价于传统写法
f = open('example.txt', 'w')
try:
    f.write("Hello, World!")
finally:
    f.close()
```

```python
def is_prime(n):
    """判断一个数是否为素数"""
    if n < 2:
        return False
    for i in range(2, int(n**0.5) + 1):
        if n % i == 0:
            return False
    return True

def generate_primes(limit):
    """生成指定范围内的所有素数"""
    primes = []
    for num in range(2, limit + 1):
        if is_prime(num):
            primes.append(num)
    return primes
```

```python
def fibonacci(n):
    """生成前n个斐波那契数"""
    fib_sequence = []
    a, b = 0, 1
    for _ in range(n):
        fib_sequence.append(a)
        a, b = b, a + b
    return fib_sequence

def fibonacci_generator():
    """斐波那契数列生成器"""
    a, b = 0, 1
    while True:
        yield a
        a, b = b, a + b

# 测试
print("前10个斐波那契数:")
```

```python
def analyze_scores(scores):
    """分析分数数据"""
    if not scores:
        return "没有数据"
    
    # 使用循环计算统计信息
    total = 0
    count = 0
    max_score = scores[0]
    min_score = scores[0]
    
    for score in scores:
        total += score
        count += 1
        if score > max_score:
            max_score = score
        if score < min_score:
            min_score = score
    
    average = total / count
    
    # 使用列表推导式找出及格和优秀的人数
    passed = len([s for s in scores if s >= 60])
    excellent = len([s for s in scores if s >= 90])
    
    return {
        "平均分": round(average, 2),
        "最高分": max_score,
        "最低分": min_score,
        "及格人数": passed,
        "优秀人数": excellent
    }

# 测试
scores = [85, 92, 78, 45, 67, 88, 95, 72, 60, 83]
analysis = analyze_scores(scores)

print("分数分析结果:")
for key, value in analysis.items():
    print(f"{key}: {value}")
# 输出：
# 平均分: 76.5
# 最高分: 95
# 最低分: 45
# 及格人数: 9
# 优秀人数: 3
```
